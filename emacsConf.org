** early-init.el
Runs before ~init.el~ and before the first frame is drawn. Disabling GUI elements here avoids the flash of default toolbars/scrollbars.
#+begin_src emacs-lisp :tangle early-init.el
  ;; -*- lexical-binding: t; -*-

  ;; Minimize garbage collection during startup
  (setq gc-cons-threshold most-positive-fixnum)

  ;; Disable GUI elements before frame is drawn
  (setq inhibit-startup-message t)
  (push '(menu-bar-lines . 0) default-frame-alist)
  (push '(tool-bar-lines . 0) default-frame-alist)
  (push '(vertical-scroll-bars) default-frame-alist)
  (push '(tooltip-mode . -1) default-frame-alist)

  ;; Prevent package.el from loading packages before init.el
  (setq package-enable-at-startup nil)
#+end_src

    '((:buffer-name "*Ilist*" :hook imenu-list-update-hook :size 40)

    "List of buffer configurations. Each entry is a plist with keys:

    - :size: The desired width of the buffer window.")

    (when-let ((window (get-buffer-window buffer-name)))

            (hook (plist-get config :hook))

* Scratch buffer
#+begin_src emacs-lisp
  ;; org-mode scratch buffer with dynamic keybindings
  (setq initial-major-mode 'org-mode)
  (setq initial-scratch-message nil)

  (defvar my/scratch-keybinds
    '((find-file            . "find and open a file")
      (org-roam-node-find   . "search org-roam notes")
      (consult-recent-file  . "open a recent file")
      (consult-buffer       . "switch between buffers"))
    "Commands and labels shown in the scratch buffer.")

  (defun my/lookup-keybind (cmd)
    "Return a human-readable key string for CMD."
    (let ((key (or (and (boundp 'evil-normal-state-map)
                        (where-is-internal cmd evil-normal-state-map t))
                   (where-is-internal cmd nil t))))
      (when key (key-description key))))

  (defun my/generate-scratch-message ()
    "Generate org-mode scratch content with dynamic keybindings."
    (let ((entries (mapcar (lambda (entry)
                             (let ((key (or (my/lookup-keybind (car entry)) "???"))
                                   (cmd-name (symbol-name (car entry))))
                               (list key cmd-name (cdr entry))))
                           my/scratch-keybinds)))
      (concat
       "\n* Scratch\n"
       (format "  /%s/\n\n" (format-time-string "%A, %d %B %Y"))
       (mapconcat
        (lambda (entry)
          (format "| ~%s~ | ~M-x %s~ | %s |"
                  (nth 0 entry) (nth 1 entry) (nth 2 entry)))
        entries "\n")
       "\n\n** Notes\n")))

  (defun my/setup-scratch-buffer ()
    "Populate the scratch buffer with org content and keybinding info."
    (with-current-buffer "*scratch*"
      (let ((inhibit-read-only t))
        (erase-buffer)
        (org-mode)
        (insert (my/generate-scratch-message))
        (goto-char (point-min))
        (while (re-search-forward "^|" nil t)
          (org-table-align))
        (goto-char (point-max))
        (set-buffer-modified-p nil))))

  (add-hook 'emacs-startup-hook #'my/setup-scratch-buffer)
